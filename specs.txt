const LanguageContext = createContext<LanguageContextType | undefined>(undefined);

export const LanguageProvider = ({ children }: { children: ReactNode }) => {
  const [language, setLanguageState] = useState<Language>(() => {
    const saved = localStorage.getItem('language');
    return (saved as Language) || 'en';
  });

  const setLanguage = (lang: Language) => {
    setLanguageState(lang);
    localStorage.setItem('language', lang);
  };

  const t = (key: string): string => {
    // Fallback logic: Try selected language -> Try English -> Return key name
    return translations[language]?.[key]  translations['en']?.[key]  key;
  };

  return (
    <LanguageContext.Provider value={{ language, setLanguage, t }}>
      {children}
    </LanguageContext.Provider>
  );
};

export const useLanguage = () => {
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error('useLanguage must be used within LanguageProvider');
  }
  return context;
};

Update useLanguage.tsx with similar code explained above.

Go through Dashboard.tsx, ProfileSettings.tsx, etc., and systematically replace strings like "Save Changes" with {t('action.save')}.

Any text that doesn't have a key in useLanguage.tsx yet will just show the key name (e.g., dashboard.something) or the fallback if you configure it that way, so make sure to add keys for every piece of text you want translated.
